# DB Entities + Schema
    1) user (user_id, name, email, password, address, phone, user_image_url, // profile_share_link)
    2) order (order_id, user_id, product_id, product_quantity, mode, payment_id)
    3) product (product_id,product_type, trending (t/f), name, company, disease_category, mrp, discount, available_stock, slug, description, images_id)
    4) image (image_id, url) [One to many]
    4) delivery (user_id, weak_id, delivery_vendor, delivery_date) [weak attribute depends on user]
    5) payment (payment_id, user_id, amount)
